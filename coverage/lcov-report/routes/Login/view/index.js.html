import React, { PureComponent } from 'react';
import PropTypes from 'prop-types';
import shallowCompare from 'react-addons-shallow-compare';
import store from 'store';
import {
  Card,
  Input,
  Button,
  Row,
  notification,
} from 'antd';
 
export default class extends PureComponent {
  static displayName = 'Login'
 
  static propTypes = {
    login: PropTypes.func,
  }
 
  state = {
    email: '',
    password: '',
  }
 
  shouldComponentUpdate(nextProps, nextState) {
    return shallowCompare(this, nextProps, nextState);
  }
 
  handleInput = e => this.setState({
    [e.target.name]: e.target.value,
  })
 
  login = () => {
    const { email, password } = this.state;
    if (email === 'marin.sokol@gmail.com' && password === 'sokol1234') {
      store.set('user', 'marin.sokol@gmail.com');
      window.location = '/';
    } else {
      notification.error({
        message: 'Neuspije≈°na prijava',
        description: 'Neispravan email ili lozinka.',
      });
    }
  }
 
  render() {
    const { email, password } = this.state;
 
    return (<div className="main-wrap login">
      <Card title="Prijava" style={{ width: 400 }}>
        <Row>
          <p>E-mail</p>
          <Input
            name="email"
            value={email}
            onChange={this.handleInput}
          />
        </Row>
        <Row>
          <p>Lozinka</p>
          <Input
            name="password"
            type="password"
            value={password}
            onChange={this.handleInput}
          />
        </Row>
        <Row>
          <Button
            type="primary"
            onClick={this.login}
          >
            Prijavi se
          </Button>
        </Row>
      </Card>
    </div>);
  }
}
